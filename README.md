	name = character 
	age = integer
	
start 
	write("Type your name: ")
	read(name)
	write("Type your age: ")
	read(age)
	

	
	if (age < 18)
	if not 
		if (age > 18) AND (age < 64) then
		write("The", name, "is over age")
		if no
		write("The", name, "is elderly")
		end if
	end if
end
